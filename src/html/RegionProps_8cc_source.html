203 std::vector<cv::Point> RegionProps::extrema()
204 {
205  std::vector<cv::Point>::iterator it;
206  std::vector<cv::Point> e;
207  cv::Point leftMost(img.cols, 0), rightMost(0,0),
208  topMost(0, 0), bottomMost(0, img.rows);
209  for(it = contour.begin(); it != contour.end(); ++it)
210  {
211  if((*it).x > rightMost.x)
212  {
213  rightMost.x = (*it).x;
214  rightMost.y = (*it).y;
215  }
216 
217  if((*it).x < leftMost.x)
218  {
219  leftMost.x = (*it).x;
220  leftMost.y = (*it).y;
221  }
222 
223  if((*it).y > topMost.y)
224  {
225  topMost.x = (*it).x;
226  topMost.y = (*it).y;
227  }
228 
229  if((*it).y < bottomMost.y)
230  {
231  bottomMost.x = (*it).x;
232  bottomMost.y = (*it).y;
233  }
234  }
235  e.push_back(rightMost);
236  e.push_back(leftMost);
237  e.push_back(topMost);
238  e.push_back(bottomMost);
239  return e;
240 }
